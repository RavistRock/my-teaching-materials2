def adjust_brightness(image_array, factor):
  """
 factor > 1 - —É–≤–µ–ª–∏—á–∏—Ç—å —è—Ä–∫–æ—Å—Ç—å
 factor < 1 - —É–º–µ–Ω—å—à–∏—Ç—å —è—Ä–∫–æ—Å—Ç—å
 """
 adjusted = np.clip(image_array.astype(np.float32) * factor, 0, 255)
 return adjusted.astype(np.uint8)
bright_image = adjust_brightness(image_array, 1.5)
dark_image = adjust_brightness(image_array, 0.7)
